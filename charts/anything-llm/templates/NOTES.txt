1. Access AnythingLLM inside the cluster:
   kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "anything-llm.fullname" . }} 3001:{{ .Values.service.port }}

2. If ingress is enabled, use one of the configured hosts:
   {{- if .Values.ingress.enabled }}
   {{- range .Values.ingress.hosts }}
   - {{ .host }}
   {{- end }}
   {{- else }}
   ingress disabled
   {{- end }}

3. If Gateway API is enabled, verify HTTPRoute admission:
   {{- if .Values.gatewayApi.enabled }}
   kubectl -n {{ .Release.Namespace }} get httproute {{ include "anything-llm.httpRouteName" . }}
   {{- else }}
   gatewayApi disabled
   {{- end }}

4. Verify application health:
   curl -fsS http://127.0.0.1:3001/api/ping
