1. Access the application inside the cluster:
   kubectl -n {{ .Release.Namespace }} port-forward svc/{{ include "coder.fullname" . }} {{ .Values.service.port }}:{{ .Values.service.port }}

2. If ingress is enabled, use one of the configured hosts:
   {{- if .Values.ingress.enabled }}
   {{- range .Values.ingress.hosts }}
   - {{ .host }}
   {{- end }}
   {{- if include "coder.ingressWildcardHost" . }}
   - {{ include "coder.ingressWildcardHost" . }}
   {{- end }}
   {{- else }}
   ingress disabled
   {{- end }}

3. If Gateway API is enabled, use one of the configured hostnames:
   {{- if .Values.gatewayApi.enabled }}
   {{- range .Values.gatewayApi.hostnames }}
   - {{ . }}
   {{- end }}
   {{- if include "coder.gatewayWildcardHost" . }}
   - {{ include "coder.gatewayWildcardHost" . }}
   {{- end }}
   {{- else }}
   gatewayApi disabled
   {{- end }}
